# useCycleList

循环浏览项目列表


## Usage

```tsx live

function Demo() {
  const [state, next, prev] = useCycleList([
    "Dog",
    "Cat",
    "Lizard",
    "Shark",
    "Whale",
    "Dolphin",
    "Octopus",
    "Seal",
  ]);

  return (
    <div>
      <div>{state}</div>
      <div>
        <button onClick={() => next()}>下一个</button>
        <button onClick={() => prev()}>上一个</button>
      </div>
    </div>
  );
};

```
