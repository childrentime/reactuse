# useDropZone

创建一个可以拖拽文件的区域

## Usage

```tsx live

function Demo() {
  const ref = useRef<HTMLDivElement>(null);

  const isOver = useDropZone(ref, (_files) => {});
  return (
    <div>
      <p>将文件拖拽到拖放区域</p>
      <div
        ref={ref}
        style={{
          minHeight: 200,
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          marginTop: "1.5rem",
          background: "rgba(156,163,175,0.1)",
        }}
      >
        <div>是否在拖放区域上方: {JSON.stringify(isOver)}</div>
      </div>
    </div>
  );
};

```
