# useAsyncEffect

支持 Promise 的 React useEffect。


## Usage

```tsx live

function Demo() {
  const [data, setData] = useState(0);

  useAsyncEffect(
    async () => {
      const result = await new Promise<number>((resolve) => {
        setTimeout(() => {
          resolve(200);
        }, 5000);
      });
      setData(result);
    },
    () => {},
    [],
  );
  return <div>數據：{data}</div>;
};

```
