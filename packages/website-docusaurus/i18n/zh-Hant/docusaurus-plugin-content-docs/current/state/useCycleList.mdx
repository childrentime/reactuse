# useCycleList

循環瀏覽項目列表


## Usage

```tsx live

function Demo() {
  const [state, next, prev] = useCycleList([
    "Dog",
    "Cat",
    "Lizard",
    "Shark",
    "Whale",
    "Dolphin",
    "Octopus",
    "Seal",
  ]);

  return (
    <div>
      <div>{state}</div>
      <div>
        <button onClick={() => next()}>下一個</button>
        <button onClick={() => prev()}>上一個</button>
      </div>
    </div>
  );
};

```
