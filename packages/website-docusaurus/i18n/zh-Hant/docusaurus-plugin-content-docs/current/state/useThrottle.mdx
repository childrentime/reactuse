# useThrottle

對值進行節流的React hook

## Usage

```tsx live

function Demo() {
  const [value, setValue] = useState<string>();
  const throttledValue = useThrottle(value, 500);

  return (
    <div>
      <input
        value={value}
        onChange={e => setValue(e.target.value)}
        placeholder="輸入值"
        style={{ width: 280 }}
      />
      <p style={{ marginTop: 16 }}>節流值：{throttledValue}</p>
    </div>
  );
};

```
